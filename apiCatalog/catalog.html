var ResponsePayload = function(code, payload) {
  this.code = code;
  this.payload = payload;
}

exports.respondWithCode = function(code, payload) {
  return new ResponsePayload(code, payload);
}

  var payload;

    code = arg2;
  }
  else {

      code = arg1;
    }
  }
  if(code && arg1) {
